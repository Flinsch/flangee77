624 static Matrix3x3 rotation_normalized(const Vector3<T>& unit_axis, T angle)
625 {
626 const Vector3<T>& u = unit_axis;
627 if (angle == T{0})
628 return IDENTITY;
629 const T cs = std::cos(angle);
630 const T sn = std::sin(angle);
631 const T C = T{1} - cs;
632 const T xsn = u.x * sn;
633 const T ysn = u.y * sn;
634 const T zsn = u.z * sn;
635 const T xC = u.x * C;
636 const T yC = u.y * C;
637 const T zC = u.z * C;
638 const T xyC = u.x * yC;
639 const T xzC = u.x * zC;
640 const T yzC = u.y * zC;
641 return {
642 u.x*xC + cs, xyC - zsn, xzC + ysn,
643 xyC + zsn, u.y*yC + cs, yzC - xsn,
644 xzC - ysn, yzC + xsn, u.z*zC + cs,
645 };
