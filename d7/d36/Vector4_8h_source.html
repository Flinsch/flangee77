411 static Vector4 cerp(const Vector4& a0, const Vector4& a, const Vector4& b, const Vector4& b1, T x)
412 {
413 const Vector4 P = (b1 - b) - (a0 - a);
414 const Vector4 Q = (a0 - a) - P;
415 const Vector4 R = b - a0;
416 const Vector4 S = a;
417 const T x2 = x*x;
418 return P * x2*x + Q * x2 + R * x + S;
