123 template <class Tstring_or_view_out, class Tstring_or_view_in>
124 requires(is_any_string_or_view_v<Tstring_or_view_out> && is_any_string_or_view_v<Tstring_or_view_in> && std::same_as<typename std::remove_cvref_t<Tstring_or_view_out>::value_type, typename std::remove_cvref_t<Tstring_or_view_in>::value_type>)
125 Tstring_or_view_out adjust_fixed_precision(Tstring_or_view_in&& s, unsigned min_precision, unsigned max_precision = MAX_FLOAT_PRECISION)
126 {
127 using Tchar = std::remove_cvref_t<Tstring_or_view_in>::value_type;
128 using Tstring_view = std::basic_string_view<Tchar>;
129 Tstring_view sv{std::forward<Tstring_or_view_in>(s)};
130
131 // Fixed notation: sign (1) + digits (34) + decimal point (1) + digits (34) = 70 characters.
132 // However, for a safe buffer size, 128 characters is often recommended
133 // in practice to account for any formatting overhead or future-proofing.
134 static constexpr size_t BUFFER_SIZE = 128;
135 static Tchar buffer[BUFFER_SIZE];
136
137 transform::trim_left(sv);
138
139 if (sv.empty())
140 return Tstring_or_view_out{};
141
142 // Skip any sign character.
143 size_t p = 0;
144 if (sv[0] == Tchar{'-'} || sv[0] == Tchar{'+'})
145 ++p;
146
147 // If, apart from a possible sign, there is
148 // only one character, it must be a digit.
149 if (sv.size() == p + 1 && !inspect::is_digit(sv[p]))
150 return Tstring_or_view_out{};
151
152 // Apart from a possible sign, the actual number must
153 // start with either a decimal point or a digit.
154 if (sv.size() > p + 1 && sv[p] != Tchar{'.'} && !inspect::is_digit(sv[p]))
155 return Tstring_or_view_out{};
156
157 // If the first character after an optional sign is a decimal
158 // point, then the following character must be a digit.
159 if (sv.size() > p + 1 && sv[p] == Tchar{'.'} && !inspect::is_digit(sv[p + 1]))
160 return Tstring_or_view_out{};
161
162 // Locate the decimal point (or its theoretical position)
163 // by skipping all digits after the optional sign.
164 size_t dot_pos = p;
165 while (dot_pos < sv.size() && inspect::is_digit(sv[dot_pos]))
166 ++dot_pos;
167
168 // Determine the total length of the (valid) number.
169 // Iterate over fractional digits, if there is a decimal point.
170 // Ignore possible decimal point if there are no fractional digits at all.
171 size_t len = dot_pos;
172 if (dot_pos + 1 < sv.size() && sv[dot_pos] == Tchar{'.'} && inspect::is_digit(sv[dot_pos + 1]))
173 for (len += 2; len < sv.size() && inspect::is_digit(sv[len]); ++len)
174 (void)0;
175
176 assert(len <= BUFFER_SIZE);
177
178 // Transfer the data to our own
179 // buffer for further processing.
180 std::memcpy(buffer, sv.data(), len);
181
182 // No digits before decimal point?
183 // Insert a leading zero.
184 if (dot_pos == p)
185 {
186 std::memmove(buffer + p + 1, buffer + p, len - p);
187 buffer[p] = Tchar{'0'};
188 ++dot_pos;
189 ++len;
190 }
191
192 // Determine the length of the fractional part.
193 size_t frac_count = len > dot_pos ? len - dot_pos - 1 : 0;
194
195 // Check the length of the fractional
196 // part against the minimum precision.
197 if (frac_count < static_cast<size_t>(min_precision))
198 {
199 // No fractional digits means there is no decimal point.
200 // Therefore, we need to add one.
201 if (frac_count == 0)
202 {
203 assert(len + 1 + min_precision <= BUFFER_SIZE);
204 buffer[len++] = Tchar{'.'};
205 }
206
207 // Now pad with zeros.
208 assert(len - frac_count + min_precision <= BUFFER_SIZE);
209 while (frac_count++ < static_cast<size_t>(min_precision))
210 buffer[len++] = Tchar{'0'};
211 } // ensure minimum precision
212
213 // "Early" out if the fractional part is
214 // not longer than the maximum precision.
215 if (frac_count <= static_cast<size_t>(max_precision))
216 return Tstring_or_view_out(Tstring_view(buffer, len));
217
218 // Determine new effective length after shortening.
219 len = dot_pos + 1 + static_cast<size_t>(max_precision);
220
221 // If we don't need to round half-up, we
222 // just need to truncate, and we're done.
223 if (buffer[len] < Tchar{'5'})
224 {
225 sv = Tstring_view(buffer, len);
226 transform::trim_right(sv, Tchar{'.'}); // Trim trailing decimal point.
227 return Tstring_or_view_out(sv);
228 }
229
230 for (size_t i = len - 1; true; --i)
231 {
232 // Digit less than 9? Then simply
233 // increase by 1 and we're done.
234 if (buffer[i] >= Tchar{'0'} && buffer[i] < Tchar{'9'})
235 {
236 ++buffer[i];
237 break;
238 }
239
240 // Do we have a 9 here? Then we have a
241 // carry-over, so we are not done yet.
242 if (buffer[i] == Tchar{'9'})
243 buffer[i] = Tchar{'0'};
244
245 // We just skip/ignore the decimal point.
246
247 // Have we reached the beginning of all the digits?
248 // If so, we need to insert a 1 here to account for the last carry-over.
249 // This will increase the length of the result by one. And that's it.
250 if (i == p)
251 {
252 std::memmove(buffer + p + 1, buffer + p, len - p);
253 buffer[p] = Tchar{'1'};
254 //++dot_pos; // We no longer need to maintain this information.
255 ++len;
256 break;
257 }
258 } // while round half-up
259
260 sv = Tstring_view(buffer, len);
261 transform::trim_right(sv, Tchar{'.'}); // Trim trailing decimal point.
262 return Tstring_or_view_out(sv);
263 }

271 template <class Tstring_or_view, typename Tval>
272 requires(is_any_string_or_view_v<Tstring_or_view> && std::is_floating_point_v<Tval>)
273 Tstring_or_view to_string(Tval val, FloatFormatOptions options)
274 {
275 using Tchar = Tstring_or_view::value_type;
276 using Tstring_view = std::basic_string_view<Tchar>;
277
278 // Fixed notation: sign (1) + digits (34) + decimal point (1) + digits (34) = 70 characters.
279 // Scientific notation: sign (1) + digit (1) + decimal point (1) + digits (34) + exponent (5) = 42 characters.
280 // However, for a safe buffer size, 128 characters is often recommended
281 // in practice to account for any formatting overhead or future-proofing.
282 static constexpr size_t BUFFER_SIZE = 128;
283 static char chars[BUFFER_SIZE];
284 static Tchar buffer[BUFFER_SIZE];
285
286 std::chars_format fmt;
287 switch (options.format)
288 {
289 case FloatFormat::General: fmt = std::chars_format::general; break;
290 case FloatFormat::Fixed: fmt = std::chars_format::fixed; break;
291 case FloatFormat::Scientific: fmt = std::chars_format::scientific; break;
292 }
293
294 // The precision parameter of `std::to_chars_result` doesn't quite behave in
295 // the way we understand precision here. In fixed/scientific modes, yes, it
296 // does mean digits after the decimal point. But in general mode, it means
297 // significant digits (total across integer plus fractional). So to control
298 // "decimal places after the dot" uniformly across formats, we need to
299 // post-process the string ourselves (see below).
300 std::to_chars_result result = std::to_chars(&chars[0], &chars[BUFFER_SIZE], val, fmt);
301
302 if (result.ec != std::errc{})
303 return Tstring_or_view{};
304
305 auto len = static_cast<size_t>(result.ptr - chars);
306
307 // Transfer character data to our buffer.
308 if constexpr (sizeof(Tchar) == sizeof(char))
309 std::memcpy(buffer, chars, len * sizeof(char));
310 else
311 std::ranges::copy(chars, chars + len, buffer);
312
313 // Always add a sign to the output?
314 if (options.force_sign && buffer[0] != Tchar{'-'} && buffer[0] != Tchar{'+'})
315 {
316 // Make space for the sign, insert
317 // it, and increase the length.
318 std::memmove(buffer + 1, buffer, len);
319 buffer[0] = Tchar{'+'};
320 ++len;
321 }
322
323 auto sv = Tstring_view(buffer, len);
324
325 // Determine the length of the fixed notation part.
326 auto fix_len = std::min(len, sv.find(Tchar{'e'}));
327
328 // Let the "helper" function perform any precision adjustments.
329 auto sv_tmp = adjust_fixed_precision<Tstring_view>(Tstring_view(buffer, fix_len), options.min_precision, options.max_precision);
330
331 // Shift the scientific notation part (if present)
332 // and copy the adjusted fixed notation part back.
333 std::memmove(buffer + sv_tmp.size(), buffer + fix_len, len - fix_len);
334 std::memcpy(buffer, sv_tmp.data(), sv_tmp.size());
335
336 // Adjust our new length(s) accordingly.
337 len -= fix_len;
338 fix_len = sv_tmp.size();
339 len += fix_len;
340
341 // There is our precision-adjusted string.
342 sv = Tstring_view(buffer, len);
343
344 // Now we only have one last (potential) problem: if an overflow occurs,
345 // resulting in a "10" at the beginning of the number in scientific notation.
346 // In that case, we need to change it to "1" and increment the exponent by 1.
347
348 // Fortunately, in this case, all the other digits are 0 anyway,
349 // so we don't need to worry about any further precision issues
350 // or other things related to the remaining digits.
351
352 // Fixed notation? Then the problem is not relevant.
353 if (len == fix_len)
354 return Tstring_or_view(sv);
355
356 // Identify the digit(s) located between the
357 // possible sign and a possible decimal point.
358 size_t p = buffer[0] == Tchar{'-'} || buffer[0] == Tchar{'+'} ? 1 : 0;
359 size_t dot_pos = std::min(fix_len, sv.find(Tchar{'.'}));
360
361 // One single digit is absolutely perfect.
362 if (dot_pos - p == 1)
363 return Tstring_or_view(sv);
364
365 // So, at this point we are dealing with
366 // said 10 in scientific notation format.
367 assert(dot_pos - p == 2);
368 assert(buffer[p + 0] == Tchar{'1'});
369 assert(buffer[p + 1] == Tchar{'0'});
370 assert(len - fix_len >= 4); // e.g., "e+02"
371
372 // Turn the 10 into a 1 by removing the 0.
373 assert(len > p + 2);
374 if (len > p + 2) // always true, but avoids underflow suspicion
375 {
376 std::memmove(buffer + p + 1, buffer + p + 2, len - (p + 2));
377 --fix_len;
378 --len;
379 }
380
381 // Update.
382 sv = Tstring_view(buffer, len);
383
384 // Now parse the exponent, increase it by 1,
385 // and write it back to our buffer.
386 signed exp = parse_int<signed>(sv.substr(fix_len + 1));
387 ++exp;
388 sv_tmp = to_string<Tstring_view>(exp, {.force_sign = true, .zero_pad_width = 2});
389 std::memcpy(buffer + fix_len + 1, sv_tmp.data(), sv_tmp.size());
390
391 // Adjust our new length accordingly.
392 len = fix_len + 1 + sv_tmp.size();
393
394 // We're finally there!
395 sv = Tstring_view(buffer, len);
396
397 return Tstring_or_view(sv);
398 }

405 template <class Tstring_or_view, typename Tval>
406 requires(is_any_string_or_view_v<Tstring_or_view> && std::is_integral_v<Tval>)
407 Tstring_or_view to_string(Tval val, IntFormatOptions options)
408 {
409 using Tchar = Tstring_or_view::value_type;
410 using Tstring_view = std::basic_string_view<Tchar>;
411
412 // Optional sign (1), up to 64-bit decimal number (20) = 21 characters.
413 static constexpr size_t BUFFER_SIZE = 1 + 20;
414 static Tchar buffer[BUFFER_SIZE];
415
416 assert(options.zero_pad_width <= 20);
417 options.zero_pad_width = std::min(options.zero_pad_width, 20u);
418
419 size_t len = 0;
420 uint64_t v;
421
422 if constexpr (std::is_unsigned_v<Tval>)
423 v = static_cast<uint64_t>(val);
424 else if (val >= 0)
425 v = static_cast<uint64_t>(val);
426 else
427 {
428 v = static_cast<uint64_t>(-val);
429 buffer[len++] = Tchar{'-'};
430 }
431
432 if (options.force_sign && len == 0)
433 buffer[len++] = Tchar{'+'};
434
435 // No prefix to consider.
436
437 const auto p = len;
438
439 do
440 {
441 auto x = static_cast<Tchar>(v % 10);
442 v /= 10;
443 Tchar c = Tchar{'0'} + x;
444 buffer[len++] = c;
445 } while (v);
446
447 while (len - p < options.zero_pad_width)
448 buffer[len++] = Tchar{'0'};
449
450 std::reverse(&buffer[p], &buffer[len]);
451
452 return Tstring_or_view(Tstring_view(buffer, len));
453 }

1008 Tval parse_float(Tstring_or_view&& s)
1009 {
1010 using Tchar = std::remove_cvref_t<Tstring_or_view>::value_type;
1011 using Tstring_view = std::basic_string_view<Tchar>;
1012 Tstring_view sv{std::forward<Tstring_or_view>(s)};
1013
1014 transform::trim(sv);
1015
1016 if (!sv.empty() && sv[0] == Tchar{'+'})
1017 sv.remove_prefix(1);
1018
1019 Tval val = {};

1021 if constexpr (std::is_same_v<Tchar, char>)
1022 {
1023 auto sv_ = sv;
1024 std::from_chars(sv_.data(), sv_.data() + sv_.size(), val);
1025 }
1026 else if constexpr (std::is_same_v<Tchar, char8_t>)
1027 {

1029 std::from_chars(sv_.data(), sv_.data() + sv_.size(), val);

1048 requires(std::is_integral_v<Tval> && is_any_string_or_view_v<Tstring_or_view>)
1049 Tval parse_int(Tstring_or_view&& s)
1050 {
1051 using Tchar = std::remove_cvref_t<Tstring_or_view>::value_type;
1052 using Tstring_view = std::basic_string_view<Tchar>;
1053 Tstring_view sv{std::forward<Tstring_or_view>(s)};
1054
1055 size_t p = inspect::count_whitespace_prefix(sv);
1056
1057 Tval val = 0;
1058 bool sign = false;
1059
1060 if (p < sv.size() && (sv[p] == Tchar{'-'} || sv[p] == Tchar{'+'}))
1061 sign = sv[p++] == Tchar{'-'};
1062
1063 if (sv.size() > p + 2 && sv[p] == Tchar{'0'})
1064 {
1065 if (sv[p + 1] == Tchar{'x'}) return parse_hex<Tval>(sv);
1066 if (sv[p + 1] == Tchar{'o'}) return parse_octal<Tval>(sv);
1067 if (sv[p + 1] == Tchar{'b'}) return parse_binary<Tval>(sv);
1068 }
1069
1070 for (; p < sv.size() && inspect::is_digit(sv[p]); ++p)
1071 {
1072 val *= 10;
1073 val += static_cast<Tval>(sv[p]) - 0x30; // '0' = 48 = 0x30
1074 }
1075
1076 return sign ? -val : val;
