118 template <any_string_or_view Tstring_or_view_out, any_string_or_view Tstring_or_view_in>
119 requires(std::same_as<typename std::remove_cvref_t<Tstring_or_view_out>::value_type, typename std::remove_cvref_t<Tstring_or_view_in>::value_type>)
120 Tstring_or_view_out adjust_fixed_precision(Tstring_or_view_in&& s, unsigned min_precision, unsigned max_precision = MAX_FLOAT_PRECISION)
121 {
122 using Tchar = std::remove_cvref_t<Tstring_or_view_in>::value_type;
123 using Tstring_view = std::basic_string_view<Tchar>;
124 Tstring_view sv{std::forward<Tstring_or_view_in>(s)};
125
126 // Fixed notation: sign (1) + digits (34) + decimal point (1) + digits (34) = 70 characters.
127 // However, for a safe buffer size, 128 characters is often recommended
128 // in practice to account for any formatting overhead or future-proofing.
129 static constexpr size_t BUFFER_SIZE = 128;
130 static Tchar buffer[BUFFER_SIZE];
131
132 transform::trim_left(sv);
133
134 if (sv.empty())
135 return Tstring_or_view_out{};
136
137 // Skip any sign character.
138 size_t p = 0;
139 if (sv[0] == Tchar{'-'} || sv[0] == Tchar{'+'})
140 ++p;
141
142 // If, apart from a possible sign, there is
143 // only one character, it must be a digit.
144 if (sv.size() == p + 1 && !inspect::is_digit(sv[p]))
145 return Tstring_or_view_out{};
146
147 // Apart from a possible sign, the actual number must
148 // start with either a decimal point or a digit.
149 if (sv.size() > p + 1 && sv[p] != Tchar{'.'} && !inspect::is_digit(sv[p]))
150 return Tstring_or_view_out{};
151
152 // If the first character after an optional sign is a decimal
153 // point, then the following character must be a digit.
154 if (sv.size() > p + 1 && sv[p] == Tchar{'.'} && !inspect::is_digit(sv[p + 1]))
155 return Tstring_or_view_out{};
156
157 // Locate the decimal point (or its theoretical position)
158 // by skipping all digits after the optional sign.
159 size_t dot_pos = p;
160 while (dot_pos < sv.size() && inspect::is_digit(sv[dot_pos]))
161 ++dot_pos;
162
163 // Determine the total length of the (valid) number.
164 // Iterate over fractional digits, if there is a decimal point.
165 // Ignore possible decimal point if there are no fractional digits at all.
166 size_t len = dot_pos;
167 if (dot_pos + 1 < sv.size() && sv[dot_pos] == Tchar{'.'} && inspect::is_digit(sv[dot_pos + 1]))
168 for (len += 2; len < sv.size() && inspect::is_digit(sv[len]); ++len)
169 (void)0;
170
171 assert(len <= BUFFER_SIZE);
172
173 // Transfer the data to our own
174 // buffer for further processing.
175 std::memcpy(buffer, sv.data(), len);
176
177 // No digits before decimal point?
178 // Insert a leading zero.
179 if (dot_pos == p)
180 {
181 std::memmove(buffer + p + 1, buffer + p, len - p);
182 buffer[p] = Tchar{'0'};
183 ++dot_pos;
184 ++len;
185 }
186
187 // Determine the length of the fractional part.
188 size_t frac_count = len > dot_pos ? len - dot_pos - 1 : 0;
189
190 // Check the length of the fractional
191 // part against the minimum precision.
192 if (frac_count < static_cast<size_t>(min_precision))
193 {
194 // No fractional digits means there is no decimal point.
195 // Therefore, we need to add one.
196 if (frac_count == 0)
197 {
198 assert(len + 1 + min_precision <= BUFFER_SIZE);
199 buffer[len++] = Tchar{'.'};
200 }
201
202 // Now pad with zeros.
203 assert(len - frac_count + min_precision <= BUFFER_SIZE);
204 while (frac_count++ < static_cast<size_t>(min_precision))
205 buffer[len++] = Tchar{'0'};
206 } // ensure minimum precision
207
208 // "Early" out if the fractional part is
209 // not longer than the maximum precision.
210 if (frac_count <= static_cast<size_t>(max_precision))
211 return Tstring_or_view_out(Tstring_view(buffer, len));
212
213 // Determine new effective length after shortening.
214 len = dot_pos + 1 + static_cast<size_t>(max_precision);
215
216 // If we don't need to round half-up, we
217 // just need to truncate, and we're done.
218 if (buffer[len] < Tchar{'5'})
219 {
220 sv = Tstring_view(buffer, len);
221 transform::trim_right(sv, Tchar{'.'}); // Trim trailing decimal point.
222 return Tstring_or_view_out(sv);
223 }
224
225 for (size_t i = len - 1; true; --i)
226 {
227 // Digit less than 9? Then simply
228 // increase by 1 and we're done.
229 if (buffer[i] >= Tchar{'0'} && buffer[i] < Tchar{'9'})
230 {
231 ++buffer[i];
232 break;
233 }
234
235 // Do we have a 9 here? Then we have a
236 // carry-over, so we are not done yet.
237 if (buffer[i] == Tchar{'9'})
238 buffer[i] = Tchar{'0'};
239
240 // We just skip/ignore the decimal point.
241
242 // Have we reached the beginning of all the digits?
243 // If so, we need to insert a 1 here to account for the last carry-over.
244 // This will increase the length of the result by one. And that's it.
245 if (i == p)
246 {
247 std::memmove(buffer + p + 1, buffer + p, len - p);
248 buffer[p] = Tchar{'1'};
249 //++dot_pos; // We no longer need to maintain this information.
250 ++len;
251 break;
252 }
253 } // while round half-up
254
255 sv = Tstring_view(buffer, len);
256 transform::trim_right(sv, Tchar{'.'}); // Trim trailing decimal point.
257 return Tstring_or_view_out(sv);
258 }

266 template <any_string_or_view Tstring_or_view, std::floating_point Tval>
267 Tstring_or_view to_string(Tval val, FloatFormatOptions options)
268 {
269 using Tchar = Tstring_or_view::value_type;
270 using Tstring_view = std::basic_string_view<Tchar>;
271
272 // Fixed notation: sign (1) + digits (34) + decimal point (1) + digits (34) = 70 characters.
273 // Scientific notation: sign (1) + digit (1) + decimal point (1) + digits (34) + exponent (5) = 42 characters.
274 // However, for a safe buffer size, 128 characters is often recommended
275 // in practice to account for any formatting overhead or future-proofing.
276 static constexpr size_t BUFFER_SIZE = 128;
277 static char chars[BUFFER_SIZE];
278 static Tchar buffer[BUFFER_SIZE];
279
280 std::chars_format fmt;
281 switch (options.format)
282 {
283 case FloatFormat::General: fmt = std::chars_format::general; break;
284 case FloatFormat::Fixed: fmt = std::chars_format::fixed; break;
285 case FloatFormat::Scientific: fmt = std::chars_format::scientific; break;
286 }
287
288 // The precision parameter of `std::to_chars_result` doesn't quite behave in
289 // the way we understand precision here. In fixed/scientific modes, yes, it
290 // does mean digits after the decimal point. But in general mode, it means
291 // significant digits (total across integer plus fractional). So to control
292 // "decimal places after the dot" uniformly across formats, we need to
293 // post-process the string ourselves (see below).
294 std::to_chars_result result = std::to_chars(&chars[0], &chars[BUFFER_SIZE], val, fmt);
295
296 if (result.ec != std::errc{})
297 return Tstring_or_view{};
298
299 auto len = static_cast<size_t>(result.ptr - chars);
300
301 // Transfer character data to our buffer.
302 if constexpr (sizeof(Tchar) == sizeof(char))
303 std::memcpy(buffer, chars, len * sizeof(char));
304 else
305 std::ranges::copy(chars, chars + len, buffer);
306
307 // Always add a sign to the output?
308 if (options.force_sign && buffer[0] != Tchar{'-'} && buffer[0] != Tchar{'+'})
309 {
310 // Make space for the sign, insert
311 // it, and increase the length.
312 std::memmove(buffer + 1, buffer, len);
313 buffer[0] = Tchar{'+'};
314 ++len;
315 }
316
317 auto sv = Tstring_view(buffer, len);
318
319 // Determine the length of the fixed notation part.
320 auto fix_len = std::min(len, sv.find(Tchar{'e'}));
321
322 // Let the "helper" function perform any precision adjustments.
323 auto sv_tmp = adjust_fixed_precision<Tstring_view>(Tstring_view(buffer, fix_len), options.min_precision, options.max_precision);
324
325 // Shift the scientific notation part (if present)
326 // and copy the adjusted fixed notation part back.
327 std::memmove(buffer + sv_tmp.size(), buffer + fix_len, len - fix_len);
328 std::memcpy(buffer, sv_tmp.data(), sv_tmp.size());
329
330 // Adjust our new length(s) accordingly.
331 len -= fix_len;
332 fix_len = sv_tmp.size();
333 len += fix_len;
334
335 // There is our precision-adjusted string.
336 sv = Tstring_view(buffer, len);
337
338 // Now we only have one last (potential) problem: if an overflow occurs,
339 // resulting in a "10" at the beginning of the number in scientific notation.
340 // In that case, we need to change it to "1" and increment the exponent by 1.
341
342 // Fortunately, in this case, all the other digits are 0 anyway,
343 // so we don't need to worry about any further precision issues
344 // or other things related to the remaining digits.
345
346 // Fixed notation? Then the problem is not relevant.
347 if (len == fix_len)
348 return Tstring_or_view(sv);
349
350 // Identify the digit(s) located between the
351 // possible sign and a possible decimal point.
352 size_t p = buffer[0] == Tchar{'-'} || buffer[0] == Tchar{'+'} ? 1 : 0;
353 size_t dot_pos = std::min(fix_len, sv.find(Tchar{'.'}));
354
355 // One single digit is absolutely perfect.
356 if (dot_pos - p == 1)
357 return Tstring_or_view(sv);
358
359 // So, at this point we are dealing with
360 // said 10 in scientific notation format.
361 assert(dot_pos - p == 2);
362 assert(buffer[p + 0] == Tchar{'1'});
363 assert(buffer[p + 1] == Tchar{'0'});
364 assert(len - fix_len >= 4); // e.g., "e+02"
365
366 // Turn the 10 into a 1 by removing the 0.
367 assert(len > p + 2);
368 if (len > p + 2) // always true, but avoids underflow suspicion
369 {
370 std::memmove(buffer + p + 1, buffer + p + 2, len - (p + 2));
371 --fix_len;
372 --len;
373 }
374
375 // Update.
376 sv = Tstring_view(buffer, len);
377
378 // Now parse the exponent, increase it by 1,
379 // and write it back to our buffer.
380 auto exp = parse_int<signed>(sv.substr(fix_len + 1));
381 ++exp;
382 sv_tmp = to_string<Tstring_view>(exp, {.force_sign = true, .zero_pad_width = 2});
383 std::memcpy(buffer + fix_len + 1, sv_tmp.data(), sv_tmp.size());
384
385 // Adjust our new length accordingly.
386 len = fix_len + 1 + sv_tmp.size();
387
388 // We're finally there!
389 sv = Tstring_view(buffer, len);
390
391 return Tstring_or_view(sv);
392 }

399 template <any_string_or_view Tstring_or_view, std::integral Tval>
400 Tstring_or_view to_string(Tval val, IntFormatOptions options)
401 {
402 using Tchar = Tstring_or_view::value_type;
403 using Tstring_view = std::basic_string_view<Tchar>;
404
405 // Optional sign (1), up to 64-bit decimal number (20) = 21 characters.
406 static constexpr size_t BUFFER_SIZE = 1 + 20;
407 static Tchar buffer[BUFFER_SIZE];
408
409 assert(options.zero_pad_width <= 20);
410 options.zero_pad_width = std::min(options.zero_pad_width, 20u);
411
412 size_t len = 0;
413 uint64_t v;
414
415 if constexpr (std::is_unsigned_v<Tval>)
416 v = static_cast<uint64_t>(val);
417 else if (val >= 0)
418 v = static_cast<uint64_t>(val);
419 else
420 {
421 v = static_cast<uint64_t>(-val);
422 buffer[len++] = Tchar{'-'};
423 }
424
425 if (options.force_sign && len == 0)
426 buffer[len++] = Tchar{'+'};
427
428 // No prefix to consider.
429
430 const auto p = len;
431
432 do
433 {
434 auto x = static_cast<Tchar>(v % 10);
435 v /= 10;
436 Tchar c = Tchar{'0'} + x;
437 buffer[len++] = c;
438 } while (v);
439
440 while (len - p < options.zero_pad_width)
441 buffer[len++] = Tchar{'0'};
442
443 std::reverse(&buffer[p], &buffer[len]);
444
445 return Tstring_or_view(Tstring_view(buffer, len));
446 }

452 Tstring_or_view to_hex(Tval val, IntFormatOptions options)
453 {
454 using Tchar = Tstring_or_view::value_type;
455 using Tstring_view = std::basic_string_view<Tchar>;
456
457 // Optional sign (1), possible "0x" prefix (2), up to 64-bit hexadecimal number (16) = 19 characters.
458 static constexpr size_t BUFFER_SIZE = 1 + 2 + 16;
459 static Tchar buffer[BUFFER_SIZE];
460
461 assert(options.zero_pad_width <= 16);
462 options.zero_pad_width = std::min(options.zero_pad_width, 16u);
463
464 size_t len = 0;
465 uint64_t v;
466
467 if constexpr (std::is_unsigned_v<Tval>)
468 v = static_cast<uint64_t>(val);
469 else if (val >= 0)
470 v = static_cast<uint64_t>(val);
471 else
472 {
473 v = static_cast<uint64_t>(-val);
474 buffer[len++] = Tchar{'-'};
475 }
476
477 if (options.force_sign && len == 0)
478 buffer[len++] = Tchar{'+'};
479
480 if (options.with_prefix)
481 {
482 buffer[len++] = Tchar{'0'};
483 buffer[len++] = Tchar{'x'};
484 }
485
486 const auto p = len;
487
488 do
489 {
490 auto x = static_cast<Tchar>(v & 0xf); // v % 16
491 v >>= 4; // v /= 16
492 Tchar c = x < 0xa ? Tchar{'0'} + x : static_cast<Tchar>(options.hex_case) + x - 0xa;
493 buffer[len++] = c;
494 } while (v);
495
496 while (len - p < options.zero_pad_width)
497 buffer[len++] = Tchar{'0'};
498
499 std::reverse(&buffer[p], &buffer[len]);
500
501 return Tstring_or_view(Tstring_view(buffer, len));

990 template <std::floating_point Tval = float, any_string_or_view Tstring_or_view>
991 Tval parse_float(Tstring_or_view&& s)
992 {
993 using Tchar = std::remove_cvref_t<Tstring_or_view>::value_type;
994 using Tstring_view = std::basic_string_view<Tchar>;
995 Tstring_view sv{std::forward<Tstring_or_view>(s)};
996
997 transform::trim(sv);
998
999 if (!sv.empty() && sv[0] == Tchar{'+'})
1000 sv.remove_prefix(1);
1001
1002 Tval val = {};
1003
1004 if constexpr (std::is_same_v<Tchar, char>)
1005 {
1006 auto sv_ = sv;
1007 std::from_chars(sv_.data(), sv_.data() + sv_.size(), val);
1008 }
1009 else if constexpr (std::is_same_v<Tchar, char8_t>)
1010 {
1011 auto sv_ = codec::reinterpret_utf8(sv);
1012 std::from_chars(sv_.data(), sv_.data() + sv_.size(), val);
1013 }

1017 auto sv_ = codec::reinterpret_utf8(make_string_view(s_));
1018 std::from_chars(sv_.data(), sv_.data() + sv_.size(), val);

1021 return val;

1030 template <std::integral Tval, any_string_or_view Tstring_or_view>
1031 Tval parse_int(Tstring_or_view&& s)
1032 {
1033 using Tchar = std::remove_cvref_t<Tstring_or_view>::value_type;
1034 using Tstring_view = std::basic_string_view<Tchar>;
1035 Tstring_view sv{std::forward<Tstring_or_view>(s)};
1036
1037 transform::trim(sv);
1038
1039 size_t p = 0;
1040 bool sign = false;
1041
1042 if (p < sv.size() && (sv[p] == Tchar{'-'} || sv[p] == Tchar{'+'}))
1043 sign = sv[p++] == Tchar{'-'};
1044
1045 if (sv.size() > p + 2 && sv[p] == Tchar{'0'})
1046 {
1047 if (sv[p + 1] == Tchar{'x'}) return parse_hex<Tval>(sv);
1048 if (sv[p + 1] == Tchar{'o'}) return parse_octal<Tval>(sv);
1049 if (sv[p + 1] == Tchar{'b'}) return parse_binary<Tval>(sv);
1050 }
1051
1052 Tval val = 0;
1053
1054 for (; p < sv.size() && inspect::is_digit(sv[p]); ++p)
1055 {
1056 val *= 10;
1057 val += static_cast<Tval>(sv[p]) - 0x30; // '0' = 48 = 0x30
1058 }
1059
1060 return sign ? -val : val;
1061 }
