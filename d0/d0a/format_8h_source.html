89 requires(std::is_integral_v<Tval> && is_any_string_or_view_v<Tstring_or_view>)
90 Tval parse_hex(Tstring_or_view&& s)
91 {
92 static constexpr auto c0 = Tchar{'0'};
93 static constexpr auto cx = Tchar{'x'};
94 std::basic_string_view<typename std::remove_cvref_t<Tstring_or_view>::value_type> sv{std::forward<Tstring_or_view>(s)};
95 Tval val = 0;
96 size_t p = inspect::count_whitespace_prefix(sv);
97 if (p + 1 < sv.size() && sv[p] == c0 && sv[p + 1] == cx)
98 p += 2;
99 for (; p < sv.size() && inspect::is_hex_digit(sv[p]); ++p)
100 {
101 val *= 0x10;
102 val += inspect::is_digit(sv[p])
103 ? static_cast<Tval>(sv[p]) - 0x30 // '0' = 48 = 0x30
104 : (static_cast<Tval>(sv[p]) | 0x20) - 0x57; // 'a' = 97 = 0x61; set off 10: 87 = 0x57
105 }
106 return val;
107 }
