113 template <typename Tchar>
114 requires(std::is_integral_v<Tchar>)
115 bool is_whitespace(Tchar c)
116 {
117 // https://www.unicode.org/Public/UCD/latest/ucd/PropList.txt
118 const auto ui32 = static_cast<uint32_t>(c);
119 return
120 (ui32 >= 0x0009 && ui32 <= 0x000d) || // tab ... carriage return
121 ui32 == 0x0020 || // space
122 ui32 == 0x0085 || // next line
123 ui32 == 0x00a0 || // no-break space
124 ui32 == 0x1680 || // Ogham space mark
125 (ui32 >= 0x2000 && ui32 <= 0x200a) || // en quad ... hair space
126 ui32 == 0x2028 || // line separator
127 ui32 == 0x2029 || // paragraph separator
128 ui32 == 0x202f || // narrow no-break space
129 ui32 == 0x205f || // medium mathematical space
130 ui32 == 0x3000; // ideographic space
131
132 static_assert(!std::is_same_v<Tchar, u8char_t>, "Not implemented for UTF-8 and its variable-length character encoding.");
133 }

570 template <class Tstring, typename Tval, typename Tchar = typename Tstring::value_type>
571 requires(is_any_string_v<Tstring> && std::is_unsigned_v<Tval> && std::is_integral_v<Tchar>)
572 Tstring to_hex(Tval val, unsigned pad_zeros = 0, Tchar ca = Tchar{'A'})
573 {
574 static constexpr auto c0 = Tchar{'0'};
575 if (val == 0)
576 return pad_zeros ? Tstring(pad_zeros, c0) : Tstring();
577 Tstring s;
578 s.reserve(18); // 16 + 2 (up to 64-bit hexadecimal number with possible prefix "0x")
579 while (val)
580 {
581 auto x = static_cast<Tchar>(val & 0xf);
582 val >>= 4;
583 Tchar c = x < 0xa ? c0+x : ca+x-0xa;
584 s += c;
585 }
586 rpad(s, pad_zeros, c0);
587 std::reverse(s.begin(), s.end());
588 return s;
589 }

646 template <class Tstring_or_view>
647 requires(is_any_string_or_view_v<Tstring_or_view>)
648 size_t levenshtein(const Tstring_or_view& s1, const Tstring_or_view& s2)
649 {
650 const size_t size1 = s1.size();
651 const size_t size2 = s2.size();
652 if (size1 == 0) return size2;
653 if (size2 == 0) return size1;
654
655 const size_t len = size2 + 1;
656 const size_t total_size = (size1 + 1) * (size2 + 1);
657
658 auto mat = std::make_unique_for_overwrite<size_t[]>(total_size);
659
660 for (size_t i = 0; i <= size1; ++i)
661 mat[i * len + 0] = i;
662 for (size_t j = 0; j <= size2; ++j)
663 mat[0 * len + j] = j;
664
665 for (size_t i = 1; i <= size1; ++i)
666 {
667 for (size_t j = 1; j <= size2; ++j)
668 {
669 const size_t cost = s1[i - 1] == s2[j - 1] ? 0 : 1;
670 mat[i * len + j] = (std::min)({
671 mat[(i - 1) * len + (j)] + 1,
672 mat[(i) * len + (j - 1)] + 1,
673 mat[(i - 1) * len + (j - 1)] + cost,
674 });
675 } // for ... j
676 } // for ... i
677
678 return mat[size1 * len + size2];
679 }

687 template <class Tstring_or_view, typename Tfloat = float>
688 requires(is_any_string_or_view_v<Tstring_or_view> && std::is_floating_point_v<Tfloat>)
689 Tfloat levenshtein_normalized(const Tstring_or_view& s1, const Tstring_or_view& s2)
