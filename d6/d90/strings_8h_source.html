86 template <typename Tchar>
87 bool is_whitespace(Tchar c)
88 {
89 // https://www.unicode.org/Public/UCD/latest/ucd/PropList.txt
90 const auto ui32 = static_cast<uint32_t>(c);
91 return
92 (ui32 >= 0x0009 && ui32 <= 0x000d) || // tab ... carriage return
93 ui32 == 0x0020 || // space
94 ui32 == 0x0085 || // next line
95 ui32 == 0x00a0 || // no-break space
96 ui32 == 0x1680 || // Ogham space mark
97 (ui32 >= 0x2000 && ui32 <= 0x200a) || // en quad ... hair space
98 ui32 == 0x2028 || // line separator
99 ui32 == 0x2029 || // paragraph separator
100 ui32 == 0x202f || // narrow no-break space
101 ui32 == 0x205f || // medium mathematical space
102 ui32 == 0x3000; // ideographic space
103
104 static_assert(!std::is_same_v<Tchar, u8char_type>, "Not implemented for UTF-8 and its variable-length character encoding.");
105 }

417 template <class Tstring, typename Tval, typename Tchar = typename Tstring::value_type>
418 requires(is_any_string_v<Tstring> && std::is_unsigned_v<Tval>)
419 Tstring to_hex(Tval val, Tchar ca = Tchar('A'), unsigned pad_zeros = 0)
420 {
421 static constexpr auto c0 = Tchar('0');
422 if (val == 0)
423 return pad_zeros ? Tstring(pad_zeros, c0) : Tstring();
424 Tstring s;
425 s.reserve(18); // 16 + 2 (up to 64-bit hexadecimal number with possible prefix "0x")
426 while (val)
427 {
428 auto x = static_cast<Tchar>(val & 0xf);
429 val >>= 4;
430 Tchar c = x < 0xa ? c0+x : ca+x-0xa;
431 s += c;
432 }
433 rpad(s, pad_zeros, c0);
434 std::reverse(s.begin(), s.end());
435 return s;
436 }

460 template <class Tstring_or_view>
461 requires(is_any_string_or_view_v<Tstring_or_view>)
462 size_t levenshtein(const Tstring_or_view& s1, const Tstring_or_view& s2)
463 {
464 const size_t size1 = s1.size();
465 const size_t size2 = s2.size();
466 if (size1 == 0) return size2;
467 if (size2 == 0) return size1;
468
469 const size_t len = size2 + 1;
470 const size_t total_size = (size1 + 1) * (size2 + 1);
471
472 auto mat = std::make_unique_for_overwrite<size_t[]>(total_size);
473
474 for (size_t i = 0; i <= size1; ++i)
475 mat[i * len + 0] = i;
476 for (size_t j = 0; j <= size2; ++j)
477 mat[0 * len + j] = j;
478
479 for (size_t i = 1; i <= size1; ++i)
480 {
481 for (size_t j = 1; j <= size2; ++j)
482 {
483 const size_t cost = s1[i - 1] == s2[j - 1] ? 0 : 1;
484 mat[i * len + j] = (std::min)({
485 mat[(i - 1) * len + (j)] + 1,
486 mat[(i) * len + (j - 1)] + 1,
487 mat[(i - 1) * len + (j - 1)] + cost,
488 });
489 } // for ... j
490 } // for ... i
491
492 return mat[size1 * len + size2];
493 }

501 template <class Tstring_or_view, typename Tfloat = float>
502 requires(is_any_string_or_view_v<Tstring_or_view> && std::is_floating_point_v<Tfloat>)
503 Tfloat levenshtein_normalized(const Tstring_or_view& s1, const Tstring_or_view& s2)
504 {
505 const size_t size = (std::max)(s1.size(), s2.size());
506 if (size == 0) return 0.0f;
507 return static_cast<Tfloat>(levenshtein(s1, s2)) / static_cast<Tfloat>(size);
508 }
