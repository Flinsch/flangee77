110 void _do_transcode(ErrorStatus& error_status, string_view_type input, string_span_type output) const
111 {
112 size_t output_offset = 0;
113
114 size_t input_offset = 0;
115 while (input_offset < input.size())
116 {
117 const auto decode_result = InputCodec::decode_one(error_status, input.substr(input_offset), *_error_handler);
118 const auto encode_result = OutputCodec::encode_one(error_status, decode_result.codepoint, output.subspan(output_offset), *_error_handler);
119 input_offset += decode_result.input_read.size();
120 output_offset += encode_result.output_written.size();
121 // Check output offset separately and not in the while condition,
122 // but explicitly after the (first) transcoding step to generate
123 // any errors regarding a supposedly empty/exhausted buffer.
124 if (output_offset >= output.size())
125 break;
126 }
127 assert(input_offset == input.size());
128 }
