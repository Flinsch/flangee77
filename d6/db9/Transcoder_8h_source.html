105 void _do_transcode(ErrorStatus& error_status, string_view_type input, string_span_type output) const
106 {
107 size_t output_offset = 0;
108
109 size_t input_offset = 0;
110 while (input_offset < input.size())
111 {
112 const auto decode_result = InputCodec::decode_one(error_status, input.substr(input_offset), *_error_handler);
113 const auto encode_result = OutputCodec::encode_one(error_status, decode_result.codepoint, output.subspan(output_offset), *_error_handler);
114 input_offset += decode_result.input_read.size();
115 output_offset += encode_result.output_written.size();
116 // Check output offset separately and not in the while condition,
117 // but explicitly after the (first) transcoding step to generate
118 // any errors regarding a supposedly empty/exhausted buffer.
119 if (output_offset >= output.size())
120 break;
121 }
122 assert(input_offset == input.size());
123 }
