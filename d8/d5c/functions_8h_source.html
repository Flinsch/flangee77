499 template <std::floating_point U = float, std::integral T>
500 constexpr U precise_divide(T numerator, T denominator)
501 {
502 if (denominator == 0)
503 {
504 if (numerator < 0)
505 return -std::numeric_limits<U>::infinity();
506 return std::numeric_limits<U>::infinity();
507 }
508
509 if (numerator == 0)
510 return U{0};
511
512 U sign = U{1};
513 if (denominator < 0)
514 {
515 sign = -sign;
516 denominator = -denominator;
517 }
518
519 T quotient = numerator / denominator;
520 T remainder = numerator % denominator;
521
522 return sign * (static_cast<U>(quotient) + static_cast<U>(remainder) / static_cast<U>(denominator));
523 }
