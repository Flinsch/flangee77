250 requires(std::is_floating_point_v<T>)
251 constexpr bool is_one_sqr(T sqr, T epsilon = std::numeric_limits<T>::epsilon())
252 {
253 // x = 1+epsilon
254 // x^2 = (1+epsilon)^2
255 // x^2 = 1^2 + 2*epsilon + epsilon^2
256 // x^2 = 1 + 2*epsilon
257 // epsilon^2 is omitted because it is far below floating point precision.
258 return std::abs(sqr - T(1)) <= T(2) * epsilon;
