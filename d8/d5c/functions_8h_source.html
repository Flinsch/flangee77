392 requires(std::is_floating_point_v<T>)
393 constexpr bool is_one_squared(T x, T epsilon = std::numeric_limits<T>::epsilon())
394 {
395 // x = 1+epsilon
396 // x^2 = (1+epsilon)^2
397 // x^2 = 1^2 + 2*epsilon + epsilon^2
398 // x^2 = 1 + 2*epsilon

400 return std::abs(x - T(1)) <= T(2) * epsilon;
