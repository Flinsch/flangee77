184 constexpr bool is_one_sqr(T sqr, T epsilon = std::numeric_limits<T>::epsilon())
185 {
186 // x = 1+epsilon
187 // x^2 = (1+epsilon)^2
188 // x^2 = 1^2 + 2*epsilon + epsilon^2
189 // x^2 = 1 + 2*epsilon

191 return std::abs(sqr - T(1)) <= T(2) * epsilon;
