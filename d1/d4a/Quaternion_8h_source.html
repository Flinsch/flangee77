90 constexpr explicit Quaternion(const ml7::Matrix3x3& m)
91 {
92 // https://www.ljll.fr/~frey/papers/scientific%20visualisation/Shoemake%20K.,%20Quaternions.pdf
93 const float t = m._11 + m._22 + m._33;
94 if (t >= 0.0f)
95 {
96 const float r = ::sqrtf(t + 1.0f);
97 const float _05i = 0.5f / r;
98 x = (m._32 - m._23) * _05i;
99 y = (m._13 - m._31) * _05i;
100 z = (m._21 - m._12) * _05i;
101 w = 0.5f * r;
102 }
103 else // => t < 0.0f
104 {
105 unsigned i = 0;
106 if (m._22 > m._11) i = 1;
107 if (m._33 > m[i][i]) i = 2;
108 unsigned j = (i + 1) % 3;
109 unsigned k = (i + 2) % 3;
110
111 const float r = ::sqrtf(m[i][i] - m[j][j] - m[k][k] + 1.0f);
112 const float _05i = 0.5f / r;
113 data[i] = 0.5f * r;
114 data[j] = (m[i][j] + m[j][i]) * _05i;
115 data[k] = (m[k][i] + m[i][k]) * _05i;
116 w = (m[k][j] - m[j][k]) * _05i;
117 }
