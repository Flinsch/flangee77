93 constexpr explicit Quaternion(const Matrix3x3<T>& m)
94 {
95 // https://www.ljll.fr/~frey/papers/scientific%20visualisation/Shoemake%20K.,%20Quaternions.pdf
96 const T t = m._11 + m._22 + m._33;
97 if (t >= T{0})
98 {
99 const T r = std::sqrt(t + T{1});
100 const T _05i = T{0.5} / r;
101 x = (m._32 - m._23) * _05i;
102 y = (m._13 - m._31) * _05i;
103 z = (m._21 - m._12) * _05i;
104 w = T{0.5} * r;
105 }
106 else // => t < T{0}
107 {
108 unsigned i = 0;
109 if (m._22 > m._11) i = 1;
110 if (m._33 > m[i][i]) i = 2;
111 unsigned j = (i + 1) % 3;
112 unsigned k = (i + 2) % 3;
113
114 const T r = std::sqrt(m[i][i] - m[j][j] - m[k][k] + T{1});
115 const T _05i = T{0.5} / r;
116 data[i] = T{0.5} * r;
117 data[j] = (m[i][j] + m[j][i]) * _05i;
118 data[k] = (m[k][i] + m[i][k]) * _05i;
119 w = (m[k][j] - m[j][k]) * _05i;
120 }
121 }

275 Vector3<T> transform(const Vector3<T>& v) const
276 {
277 Vector3<T> u{x, y, z};
278 Vector3<T> uv = u.cross(v);
279 Vector3<T> uuv = u.cross(uv);
280 uv *= T{2} * w;
281 uuv *= T{2};
282 return v + uv + uuv;
283 }
284
289 Vector3<T> transform_inverted(const Vector3<T>& v) const
290 {
291 T d = norm();
292 if (d == T{0})
293 return v;
294 d = T{-1} / d;
295 Vector3<T> u{x*d, y*d, z*d};
296 Vector3<T> uv = u.cross(v);
297 Vector3<T> uuv = u.cross(uv);
298 uv *= T{2} * -w*d;
299 uuv *= T{2};
300 return v + uv + uuv;
301 }
