39 // Step 1: Sign change detection
40 for (unsigned k = 0; k < subdivisions; ++k)
41 {
42 const T x0 = lower_bound + static_cast<T>(k) * step;
43 const T x1 = x0 + step;
44 const T f0 = f(x0);
45 const T f1 = f(x1);
46
47 T x; // This will be our root value, if found, later.
48

50 const bool sign_change = std::signbit(f0) != std::signbit(f1);
51 if (sign_change)

53 // Take the midpoint as initial guess.
54 //x = (x0 + x1) * T{0.5};
55 // No, interpret this subdivision as a
56 // straight line and take its 0-intercept.
57 // y = ax + b
58 // a = (y2 - y1) / (x2 - x1)
59 // b = y1 - a * x1
60 // x = -b / a
61 // = x1 - y1 * (x2 - x1) / (y2 - y1)

63 // It's actually not that much more expensive and is definitely
64 // worth the increased (starting) precision we get from it.
65 }

72 x = x1;

91 // Otherwise we might get too many apparent roots
92 // for a whole series (possibly all) subdivisions.

104 break;
105 }
106 } // for each Newton-Raphson iteration

110 // Otherwise: converged?
