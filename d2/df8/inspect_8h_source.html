15 template <any_char Tchar>
16 bool is_whitespace(Tchar c)
17 {
18 // https://www.unicode.org/Public/UCD/latest/ucd/PropList.txt
19 const auto ui32 = static_cast<uint32_t>(c);
20 return
21 (ui32 >= 0x0009 && ui32 <= 0x000d) || // tab ... carriage return
22 ui32 == 0x0020 || // space
23 ui32 == 0x0085 || // next line
24 ui32 == 0x00a0 || // no-break space
25 ui32 == 0x1680 || // Ogham space mark
26 (ui32 >= 0x2000 && ui32 <= 0x200a) || // en quad ... hair space
27 ui32 == 0x2028 || // line separator
28 ui32 == 0x2029 || // paragraph separator
29 ui32 == 0x202f || // narrow no-break space
30 ui32 == 0x205f || // medium mathematical space
31 ui32 == 0x3000; // ideographic space
32
33 static_assert(!std::is_same_v<Tchar, u8char_t>, "Not implemented for UTF-8 and its variable-length character encoding.");
34 }
