16 requires(std::is_integral_v<Tchar>)
17 bool is_whitespace(Tchar c)
18 {
19 // https://www.unicode.org/Public/UCD/latest/ucd/PropList.txt
20 const auto ui32 = static_cast<uint32_t>(c);
21 return
22 (ui32 >= 0x0009 && ui32 <= 0x000d) || // tab ... carriage return
23 ui32 == 0x0020 || // space
24 ui32 == 0x0085 || // next line
25 ui32 == 0x00a0 || // no-break space
26 ui32 == 0x1680 || // Ogham space mark
27 (ui32 >= 0x2000 && ui32 <= 0x200a) || // en quad ... hair space
28 ui32 == 0x2028 || // line separator
29 ui32 == 0x2029 || // paragraph separator
30 ui32 == 0x202f || // narrow no-break space
31 ui32 == 0x205f || // medium mathematical space
32 ui32 == 0x3000; // ideographic space
33
34 static_assert(!std::is_same_v<Tchar, u8char_t>, "Not implemented for UTF-8 and its variable-length character encoding.");
35 }
